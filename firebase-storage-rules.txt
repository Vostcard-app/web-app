// Firebase Storage Security Rules for Vostcard App
// Copy these rules to your Firebase Console > Storage > Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow authenticated users to upload their own vostcards
    match /vostcards/{userId}/{vostcardId}/{fileName} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow public read access to posted vostcards (for map display)
    match /vostcards/{userId}/{vostcardId}/{fileName} {
      allow read: if true; // Anyone can read posted content
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow authenticated users to upload their own private vostcards
    match /privateVostcards/{userId}/{vostcardId}/{fileName} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow users to upload profile pictures
    match /users/{userId}/profile.jpg {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow public read access to user avatars
    match /users/{userId}/avatar.jpg {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Default: deny all other access
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
